!create a list for active quests
$quest_list = ''

n = arrsize('$quest') - 1
i = 0
:loop
    i += 1
    if active[$quest[i]] = 1: $quest_list += '   <<$quest[i]>><br>'
if i < n: jump 'loop'

if quest_description_mode = 1: $quest_list = '   <<$quest_description[choosen_quest]>>' 
--- $quest_list ---------------------------------
