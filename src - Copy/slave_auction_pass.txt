# slave_auction_pass
!Skip over one auction phase (buying slave)
IF auction_phase = 5:
 haggling_stage = 0
 auction_phase = 7
 GT 'city_menu_form'
END

IF auction_phase = 7:
 haggling_stage = 0
 GT 'city_menu_form'
END

IF auction_phase = 6:
 sparks -= cur_stock_price
 haggling_stage = 0 
 auction_phase = 7
 GT 'stock_to_slave'
END

IF auction_phase = 4:
 IF stock_sold => stock_limit:
  auction_phase = 5
  else
  IF haggling_stage = 1:
   auction_phase = 0
   haggling_stage = 0
   gs 'generate_stock'  
  END
  rises = 0  
 END
 IF haggling_stage = 2: 
  auction_phase = 6 
  gt 'slave_auction_buy'
 END
END

IF  haggling_stage > 0:
 stake_size = 0
 :loop
 rise_chance = RAND (1,haggling_limit)
 rise_chance += rises 
 IF rise_chance <= haggling_limit: 
  rises += 1
  stake_size += 1
  if stake_size < 10: jump 'loop'
 END 
 IF stake_size = 0:
  if auction_phase < 4: auction_phase += 1
  else
  cur_stock_price += stake_size * 5
  rise_stock_price = cur_stock_price + 5
  a = RAND (1,35)
  $text = '<<$rival_stake[a]>> <<$stake_size[stake_size]>>'
  haggling_stage = 1  
  auction_phase = 1  
 END
END

IF haggling_stage = 0: 
 stock_sold += 1
 cur_stock_price = stock_base_price
 rise_stock_price = stock_base_price 
 haggling_stage = 1
 else
 rise_stock_price = cur_stock_price + 5
END

GT 'slave_auction_buy'
