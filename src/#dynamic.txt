 IF hero_points >= 80:
  master_fetish = 5
  hero_points -= 80
  else
  msg "Not enough character points. You need 80 points to raise this stat to the fifth level."
 END
END
IF master_fetish = 3:
 IF hero_points >= 40:
  master_fetish = 4
  hero_points -= 40
  else
  msg "Not enough character points. You need 40 points to raise this stat to the fourth level."
 END
END
IF master_fetish = 2:
 IF hero_points >= 20:
  master_fetish = 3
  hero_points -= 20
  else
  msg "Not enough character points. You need 20 points to raise this stat to the third level."
 END
END
IF master_fetish = 1:
 IF hero_points >= 10:
  master_fetish = 2
  hero_points -= 10
  else
  msg "Not enough character points. You need 10 points to raise this stat to the second level."
 END
END
IF master_fetish = 0:
 IF hero_points >= 10:
  master_fetish = 1
  hero_points -= 10
  else
  msg "Not enough character points. You need 10 points to raise this stat to the first level."
 END
END
}

$lower_fetish = {
IF master_fetish = 1:
 master_fetish = 0
 hero_points += 10
END
IF master_fetish = 2:
 master_fetish = 1
 hero_points += 10
END
IF master_fetish = 3:
 master_fetish = 2
 hero_points += 20
END
IF master_fetish = 4:
 master_fetish = 3
 hero_points += 40
END
IF master_fetish = 5:
 master_fetish = 4
 hero_points += 80
END
}
